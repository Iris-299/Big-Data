# Extending stack interface
> Implement a stack supporting the operations Push(), Pop(), and Max().

* **Output Format**

## My Solution

#python3
import sys

class StackWithMax():
    def __init__(self):
        self.__stack = []
        self.__max = []

    def Push(self, a):
        self.__stack.append(a)
        if len(self.__max) == 0:
            self.__max.append(a)
        else:
            if self.__max[-1]>a:
                self.__max.append(self.__max[-1])
            else:
                self.__max.append(a)

    def Pop(self):
        assert(len(self.__stack))
        self.__max.pop()

    def Max(self):
        assert(len(self.__stack))
        return self.__max[-1]

if __name__ == '__main__':
    stack = StackWithMax()

    num_queries = int(input())
    for _ in range(num_queries):
        query = input().split()

        if query[0] == "push":
            stack.Push(int(query[1]))
        elif query[0] == "pop":
            stack.Pop()
        elif query[0] == "max":
            print(stack.Max())
        else:
            assert(0)
